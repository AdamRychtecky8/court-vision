% Defer \caption inside longtable so it prints at the end instead of where it's called.
\makeatletter
\let\LT@OldCaption\caption
\AtBeginEnvironment{longtable}{%
  \let\LT@DeferredCaption\@empty
  \renewcommand{\caption}{%
    \protected@edef\LT@DeferredCaption{\noexpand\LT@OldCaption}%
    \@bsphack
    \@dblarg{\LT@stash@caption}%
  }%
}
\newcommand{\LT@stash@caption}[2][]{%
  \gdef\LT@stash@opt{#1}%
  \gdef\LT@stash@txt{#2}%
  \@esphack
}
\AtEndEnvironment{longtable}{%
  \ifx\LT@stash@txt\@empty\else
    \ifx\LT@stash@opt\@empty
      \LT@OldCaption{\LT@stash@txt}%
    \else
      \LT@OldCaption[\LT@stash@opt]{\LT@stash@txt}%
    \fi
    \global\let\LT@stash@txt\@empty
  \fi
}
\makeatother
